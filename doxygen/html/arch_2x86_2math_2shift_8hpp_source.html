40WJR_ALL_NONNULL WJR_INTRINSIC_INLINE void builtin_lshift_n_impl(T *dst, const T *src, size_t n,
41 unsigned int cl) noexcept {
42 if (n <= shift_detail::small_shift_n_threshold) {
43 if (WJR_LIKELY(--n)) {
44 #if WJR_HAS_FEATURE(GCC_STYLE_INLINE_ASM) && defined(__BMI2__) && \
45 !WJR_HAS_FEATURE(FORCE_NOVECTOR)
46 unsigned int tcl = 64 - cl;
47 uint64_t __cl = cl;
48 uint64_t __tcl = tcl;
49 uint64_t t0, t1;
50 asm volatile(
51 ".Lloop%=:\n\t"
52 "shlx{q %[cl], (%[src], %[n], 8), %[t0]| %[t0], [%[src] + %[n] * 8], %[cl]}\n\t"
53 "shrx{q %[tcl], -8(%[src], %[n], 8), %[t1]| %[t1], [%[src] + %[n] * 8 - 8], %[tcl]}\n\t"
54 "or{q %[t1], %[t0]| %[t0], %[t1]}\n\t"
55 "mov{q %[t0], (%[dst], %[n], 8)| [%[dst] + %[n] * 8], %[t0]}\n\t"
56 // todo : Maybe need to use `sub'.
57 "dec %[n]\n\t"
58 "jne .Lloop%=\n\t"
59 : [dst] "+r"(dst), [src] "+r"(src), [n] "+&r"(n), [t0] "=&r"(t0), [t1] "=&r"(t1)
60 : [cl] "r"(__cl), [tcl] "r"(__tcl)
61 : "cc", "memory");
62 #else
63 WJR_NOVECTOR
64 do {
65 dst[n] = shld(src[n], src[n - 1], cl);
66 } while (WJR_LIKELY(--n));
67 #endif
68 }

72 src += n;
73 dst += n;

107WJR_ALL_NONNULL WJR_INTRINSIC_INLINE void builtin_rshift_n_impl(T *dst, const T *src, size_t n,
108 unsigned int cl) noexcept {
109 if (n <= shift_detail::small_shift_n_threshold) {
110 if (WJR_LIKELY(--n)) {
111 #if WJR_HAS_FEATURE(GCC_STYLE_INLINE_ASM) && defined(__BMI2__) && \
112 !WJR_HAS_FEATURE(FORCE_NOVECTOR)
113 unsigned int tcl = 64 - cl;
114 uint64_t __cl = cl;
115 uint64_t __tcl = tcl;
116 uint64_t t0, t1;
117 size_t i = 0;
118 asm volatile(
119 ".Lloop%=:\n\t"
120 "shrx{q %[cl], (%[src], %[i], 8), %[t0]| %[t0], [%[src] + %[i] * 8], %[cl]}\n\t"
121 "shlx{q %[tcl], 8(%[src], %[i], 8), %[t1]| %[t1], [%[src] + %[i] * 8 + 8], %[tcl]}\n\t"
122 "or{q %[t1], %[t0]| %[t0], %[t1]}\n\t"
123 "mov{q %[t0], (%[dst], %[i], 8)| [%[dst] + %[i] * 8], %[t0]}\n\t"
124 "add{q $1, %[i]| %[i], 1}\n\t"
125 "cmp{q %[i], %[n]| %[n], %[i]}\n\t"
126 "jne .Lloop%=\n\t"
127 : [dst] "+r"(dst), [src] "+r"(src), [i] "+&r"(i), [t0] "=&r"(t0), [t1] "=&r"(t1)
128 : [cl] "r"(__cl), [tcl] "r"(__tcl), [n] "r"(n)
129 : "cc", "memory");
130 #else
131 WJR_NOVECTOR
132 do {
133 dst[0] = shrd(src[0], src[1], cl);
134 ++dst;
135 ++src;
136 } while (WJR_LIKELY(--n));
137 #endif
138 }
139
140 return;
141 }

143 if (WJR_BUILTIN_CONSTANT_P(cl)) {
144 return large_builtin_rshift_n_impl<true>(dst, src, n - 1, cl);
