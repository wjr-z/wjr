33WJR_INTRINSIC_INLINE int64_t rdtsc() noexcept {
34#if defined(__i386__)
35 int64_t ret;
36 __asm__ volatile("rdtsc" : "=A"(ret));
37 return ret;
38#elif defined(__x86_64__) || defined(__amd64__)
39 uint64_t low, high;
40 __asm__ volatile("rdtsc" : "=a"(low), "=d"(high));
41 return static_cast<int64_t>((high << 32) | low);
42#elif defined(__powerpc__) || defined(__ppc__)
43 // This returns a time-base, which is not always precisely a cycle-count.
44 #if defined(__powerpc64__) || defined(__ppc64__)
45 int64_t tb;
46 asm volatile("mfspr %0, 268" : "=r"(tb));
47 return tb;
48 #else
49 uint32_t tbl, tbu0, tbu1;
50 asm volatile(
51 "mftbu %0\n"
52 "mftb %1\n"
53 "mftbu %2"
54 : "=r"(tbu0), "=r"(tbl), "=r"(tbu1));
55 tbl &= -static_cast<int32_t>(tbu0 == tbu1);
56 // high 32 bits in tbu1; low 32 bits in tbl (tbu0 is no longer needed)
57 return (static_cast<uint64_t>(tbu1) << 32) | tbl;
58 #endif
59#elif defined(__sparc__)
60 int64_t tick;
61 asm(".byte 0x83, 0x41, 0x00, 0x00");
62 asm("mov %%g1, %0" : "=r"(tick));
63 return tick;
64#elif defined(__ia64__)
65 int64_t itc;
66 asm("mov %0 = ar.itc" : "=r"(itc));
67 return itc;
68#elif defined(WJR_COMPILER_MSVC) && defined(_M_IX86)
69 // Older MSVC compilers (like 7.x) don't seem to support the
70 // __rdtsc intrinsic properly, so I prefer to use _asm instead
71 // when I know it will work. Otherwise, I'll use __rdtsc and hope
72 // the code is being compiled with a non-ancient compiler.
73 _asm rdtsc
74#elif defined(WJR_COMPILER_MSVC) && (defined(_M_ARM64) || defined(_M_ARM64EC))
75 // See // https://docs.microsoft.com/en-us/cpp/intrinsics/arm64-intrinsics
76 // and https://reviews.llvm.org/D53115
77 int64_t virtual_timer_value;
78 virtual_timer_value = _ReadStatusReg(ARM64_CNTVCT);
79 return virtual_timer_value;
80#elif defined(WJR_COMPILER_MSVC)
81 return __rdtsc();
82#elif defined(WJR_OS_NACL)
83 // Native Client validator on x86/x86-64 allows RDTSC instructions,
84 // and this case is handled above. Native Client validator on ARM
85 // rejects MRC instructions (used in the ARM-specific sequence below),
86 // so we handle it here. Portable Native Client compiles to
87 // architecture-agnostic bytecode, which doesn't provide any
88 // cycle counter access mnemonics.
89
90 // Native Client does not provide any API to access cycle counter.
91 // Use clock_gettime(CLOCK_MONOTONIC, ...) instead of gettimeofday
92 // because is provides nanosecond resolution (which is noticeable at
93 // least for PNaCl modules running on x86 Mac & Linux).
94 // Initialize to always return 0 if clock_gettime fails.
95 struct timespec ts = {0, 0};
96 clock_gettime(CLOCK_MONOTONIC, &ts);
97 return static_cast<int64_t>(ts.tv_sec) * 1000000000 + ts.tv_nsec;
98#elif defined(__aarch64__)
99 // System timer of ARMv8 runs at a different frequency than the CPU's.
100 // The frequency is fixed, typically in the range 1-50MHz. It can be
101 // read at CNTFRQ special register. We assume the OS has set up
102 // the virtual timer properly.
103 int64_t virtual_timer_value;
104 asm volatile("mrs %0, cntvct_el0" : "=r"(virtual_timer_value));
105 return virtual_timer_value;
106#elif defined(__ARM_ARCH)
107 // V6 is the earliest arch that has a standard cyclecount
108 // Native Client validator doesn't allow MRC instructions.
109 #if (__ARM_ARCH >= 6)
110 uint32_t pmccntr;
111 uint32_t pmuseren;
112 uint32_t pmcntenset;
113 // Read the user mode perf monitor counter access permissions.
114 asm volatile("mrc p15, 0, %0, c9, c14, 0" : "=r"(pmuseren));
115 if (pmuseren & 1) { // Allows reading perfmon counters for user mode code.
116 asm volatile("mrc p15, 0, %0, c9, c12, 1" : "=r"(pmcntenset));
117 if (pmcntenset & 0x80000000ul) { // Is it counting?
118 asm volatile("mrc p15, 0, %0, c9, c13, 0" : "=r"(pmccntr));
119 // The counter is set up to count every 64th cycle
120 return static_cast<int64_t>(pmccntr) * 64; // Should optimize to << 6
121 }
122 }
123 #endif
124 struct timeval tv;
125 gettimeofday(&tv, nullptr);
126 return static_cast<int64_t>(tv.tv_sec) * 1000000 + tv.tv_usec;
127#elif defined(__mips__) || defined(__m68k__)
128 // mips apparently only allows rdtsc for superusers, so we fall
129 // back to gettimeofday. It's possible clock_gettime would be better.
130 struct timeval tv;
131 gettimeofday(&tv, nullptr);
132 return static_cast<int64_t>(tv.tv_sec) * 1000000 + tv.tv_usec;
133#elif defined(__loongarch__) || defined(__csky__)
134 struct timeval tv;
135 gettimeofday(&tv, nullptr);
136 return static_cast<int64_t>(tv.tv_sec) * 1000000 + tv.tv_usec;
137#elif defined(__s390__) // Covers both s390 and s390x.
138 // Return the CPU clock.
139 uint64_t tsc;
140 #if defined(WJR_OS_ZOS) && defined(COMPILER_IBMXL)
141 // z/OS XL compiler HLASM syntax.
142 asm(" stck %0" : "=m"(tsc) : : "cc");
143 #else
144 asm("stck %0" : "=Q"(tsc) : : "cc");
145 #endif
146 return tsc;
147#elif defined(__riscv) // RISC-V
148 // Use RDTIME (and RDTIMEH on riscv32).
149 // RDCYCLE is a privileged instruction since Linux 6.6.
150 #if __riscv_xlen == 32
151 uint32_t cycles_lo, cycles_hi0, cycles_hi1;
152 // This asm also includes the PowerPC overflow handling strategy, as above.
153 // Implemented in assembly because Clang insisted on branching.
154 asm volatile(
155 "rdtimeh %0\n"
156 "rdtime %1\n"
157 "rdtimeh %2\n"
158 "sub %0, %0, %2\n"
159 "seqz %0, %0\n"
160 "sub %0, zero, %0\n"
161 "and %1, %1, %0\n"
162 : "=r"(cycles_hi0), "=r"(cycles_lo), "=r"(cycles_hi1));
163 return (static_cast<uint64_t>(cycles_hi1) << 32) | cycles_lo;
164 #else
165 uint64_t cycles;
166 asm volatile("rdtime %0" : "=r"(cycles));
167 return cycles;
168 #endif
169#elif defined(__e2k__) || defined(__elbrus__)
170 struct timeval tv;
171 gettimeofday(&tv, nullptr);
172 return static_cast<int64_t>(tv.tv_sec) * 1000000 + tv.tv_usec;
173#elif defined(__hexagon__)
174 uint64_t pcycle;
175 asm volatile("%0 = C15:14" : "=r"(pcycle));
176 return static_cast<double>(pcycle);
177#elif defined(__alpha__)
178 // Alpha has a cycle counter, the PCC register, but it is an unsigned 32-bit
179 // integer and thus wraps every ~4s, making using it for tick counts
180 // unreliable beyond this time range. The real-time clock is low-precision,
181 // roughtly ~1ms, but it is the only option that can reasonable count
182 // indefinitely.
183 struct timeval tv;
184 gettimeofday(&tv, nullptr);
185 return static_cast<int64_t>(tv.tv_sec) * 1000000 + tv.tv_usec;
186#else
187 // The soft failover to a generic implementation is automatic only for ARM.
188 // For other platforms the developer is expected to make an attempt to create
189 // a fast implementation and use generic version if nothing better is
190 // available.
191 #undef WJR_HAS_FEATURE_RDTSC
192 // #error You need to define CycleTimer for your OS and CPU
193 WJR_UNREACHABLE();
194#endif
195}
