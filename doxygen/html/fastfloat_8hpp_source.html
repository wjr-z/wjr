130struct adjusted_mantissa {
131 uint64_t mantissa{0};
132 int32_t power2{0}; // a negative value indicates an invalid result

134 return mantissa == o.mantissa && power2 == o.power2;
135 }
136 WJR_INTRINSIC_CONSTEXPR bool operator!=(const adjusted_mantissa &o) const noexcept {
137 return mantissa != o.mantissa || power2 != o.power2;
138 }

388template <typename T>
389WJR_INTRINSIC_INLINE void to_float(bool negative, adjusted_mantissa am, T &value) {
390 using fastfloat_uint = typename binary_format<T>::equiv_uint;
391 fastfloat_uint word = (fastfloat_uint)am.mantissa;
392 word |= fastfloat_uint(am.power2) << binary_format<T>::mantissa_explicit_bits();
393 word |= fastfloat_uint(negative) << binary_format<T>::sign_index();
394 value = bit_cast<T>(word);
395}

1089// This will compute or rather approximate w * 5**q and return a pair of 64-bit
1090// words approximating the result, with the high" part corresponding to the most
1091// significant bits and the low part corresponding to the least significant
1092// bits.
1093//
1094template <int bit_precision>
1095WJR_INTRINSIC_INLINE uint128_t compute_product_approximation(int64_t q, uint64_t w) {
1096 const int index = 2 * int(q - powers::smallest_power_of_five);
1097 // For small values of q, e.g., q in [0,27], the answer is always exact
1098 // because The line uint128_t firstproduct = mul64x64to128(w,
1099 // power_of_five_128[index]); gives the exact answer.
1100 uint128_t firstproduct = mul64x64to128(w, powers::power_of_five_128[index]);
1101 static_assert((bit_precision >= 0) && (bit_precision <= 64), " precision should be in (0,64]");
1102 constexpr uint64_t precision_mask = (bit_precision < 64)
1103 ? (uint64_t(0xFFFFFFFFFFFFFFFF) >> bit_precision)
1104 : uint64_t(0xFFFFFFFFFFFFFFFF);
1105 if ((firstproduct.high & precision_mask) ==
1106 precision_mask) { // could further guard with (lower + w < lower)
1107 // regarding the second product, we only need secondproduct.high, but
1108 // our expectation is that the compiler will optimize this extra work
1109 // away if needed.
1110 const uint128_t secondproduct = mul64x64to128(w, powers::power_of_five_128[index + 1]);
1111 firstproduct.low += secondproduct.high;
1112 if (secondproduct.high > firstproduct.low) {
1113 firstproduct.high++;
1114 }
1115 }
1116 return firstproduct;
1117}

1196

1198 compute_product_approximation<binary::mantissa_explicit_bits() + 3>(q, w);
1199 if (product.low == 0xFFFFFFFFFFFFFFFF) { // could guard it further
1200 // In some very rare cases, this could happen, in which case we might
1201 // need a more accurate computation that what we can provide cheaply.
1202 // This is very, very unlikely.
1203 //
1204 const bool inside_safe_exponent =
1205 (q >= -27) && (q <= 55); // always good because 5**q <2**128 when q>=0,
1206 // and otherwise, for q<0, we have 5**-q<2**64 and the 128-bit
1207 // reciprocal allows for exact computation.
1208 if (!inside_safe_exponent) {
1209 return compute_error_scaled<binary>(q, product.high, lz);
1210 }
1211 }
1212 // The "compute_product_approximation" function can be slightly slower than
1213 // a branchless approach: uint128_t product = compute_product(q, w); but in
1214 // practice, we can win big with the compute_product_approximation if its
1215 // additional branch is easily predicted. Which is best is data specific.
1216 const int upperbit = int(product.high >> 63);
1217
1218 answer.mantissa = product.high >> (upperbit + 64 - binary::mantissa_explicit_bits() - 3);
1219
1220 answer.power2 = int32_t(detail::power(int32_t(q)) + upperbit - lz - binary::minimum_exponent());
1221 if (answer.power2 <= 0) { // we have a subnormal?
1222 // Here have that answer.power2 <= 0 so -answer.power2 >= 0
1223 if (-answer.power2 + 1 >= 64) { // if we have more than 64 bits below the minimum
1224 // exponent, you have a zero for sure.
1225 answer.power2 = 0;
1226 answer.mantissa = 0;
1227 // result should be zero
1228 return answer;
1229 }
1230 // next line is safe because -answer.power2 + 1 < 64
1231 answer.mantissa >>= -answer.power2 + 1;
1232 // Thankfully, we can't have both "round-to-even" and subnormals because
1233 // "round-to-even" only occurs for powers close to 0.
1234 answer.mantissa += (answer.mantissa & 1); // round up
1235 answer.mantissa >>= 1;
1236 // There is a weird scenario where we don't have a subnormal but just.
1237 // Suppose we start with 2.2250738585072013e-308, we end up
1238 // with 0x3fffffffffffff x 2^-1023-53 which is technically subnormal
1239 // whereas 0x40000000000000 x 2^-1023-53 is normal. Now, we need to
1240 // round up 0x3fffffffffffff x 2^-1023-53 and once we do, we are no
1241 // longer subnormal, but we can only know this after rounding. So we
1242 // only declare a subnormal if we are smaller than the threshold.
1243 answer.power2 =
1244 (answer.mantissa < (uint64_t(1) << binary::mantissa_explicit_bits())) ? 0 : 1;
1245 return answer;
1246 }
1247
1248 // usually, we round *up*, but if we fall right in between and and we have
1249 // an even basis, we need to round down We are only concerned with the cases
1250 // where 5**q fits in single 64-bit word.
1251 if ((product.low <= 1) && (q >= binary::min_exponent_round_to_even()) &&
1252 (q <= binary::max_exponent_round_to_even()) &&
1253 ((answer.mantissa & 3) == 1)) { // we may fall between two floats!
1254 // To be in-between two floats we need that in doing
1255 // answer.mantissa = product.high >> (upperbit + 64 -
1256 // binary::mantissa_explicit_bits() - 3);
1257 // ... we dropped out only zeroes. But if this happened, then we can go
1258 // back!!!
1259 if ((answer.mantissa << (upperbit + 64 - binary::mantissa_explicit_bits() - 3)) ==
1260 product.high) {
1261 answer.mantissa &= ~uint64_t(1); // flip it so that we do not round up
1262 }
1263 }
1264
1265 answer.mantissa += (answer.mantissa & 1); // round up
1266 answer.mantissa >>= 1;
1267 if (answer.mantissa >= (uint64_t(2) << binary::mantissa_explicit_bits())) {
1268 answer.mantissa = (uint64_t(1) << binary::mantissa_explicit_bits());
1269 answer.power2++; // undo previous addition
1270 }
1271
1272 answer.mantissa &= ~(uint64_t(1) << binary::mantissa_explicit_bits());
1273 if (answer.power2 >= binary::infinite_power()) { // infinity
1274 answer.power2 = binary::infinite_power();
1275 answer.mantissa = 0;
1276 }
1277 return answer;
1278}

1387template <typename T>
1388from_chars_result<> parse_infnan(const char *first, const char *last, T &value) noexcept {
1389 from_chars_result<> answer;
1390 answer.ptr = first;
1391 answer.ec = std::errc(); // be optimistic
1392 bool minusSign = false;
1393 if (*first == '-') { // assume first < last, so dereference without checks;
1394 // C++17 20.19.3.(7.1) explicitly forbids '+' here
1395 minusSign = true;
1396 ++first;
1397 }
1398 if (last - first >= 3) {
1399 if (constant_length_strncaseequal(first, "nan", 3_u)) {
1400 answer.ptr = (first += 3);
1401 value = minusSign ? -std::numeric_limits<T>::quiet_NaN()
1402 : std::numeric_limits<T>::quiet_NaN();
1403 // Check for possible nan(n-char-seq-opt), C++17 20.19.3.7,
1404 // C11 7.20.1.3.3. At least MSVC produces nan(ind) and nan(snan).
1405 if (first != last && *first == '(') {
1406 for (const char *ptr = first + 1; ptr != last; ++ptr) {
1407 if (*ptr == ')') {
1408 answer.ptr = ptr + 1; // valid nan(n-char-seq-opt)
1409 break;
1410 } else if (!(('a' <= *ptr && *ptr <= 'z') || ('A' <= *ptr && *ptr <= 'Z') ||
1411 ('0' <= *ptr && *ptr <= '9') || *ptr == '_'))
1412 break; // forbidden char, not nan(n-char-seq-opt)
1413 }
1414 }
1415 return answer;
1416 }
1417 if (constant_length_strncaseequal(first, "inf", 3_u)) {
1418 if ((last - first >= 8) && constant_length_strncaseequal(first, "infinity", 8_u)) {
1419 answer.ptr = first + 8;
1420 } else {
1421 answer.ptr = first + 3;
1422 }
1423 value = minusSign ? -std::numeric_limits<T>::infinity()
1424 : std::numeric_limits<T>::infinity();
1425 return answer;
1426 }
1427 }
1428 answer.ec = std::errc::invalid_argument;
1429 return answer;
1430}

1437WJR_INTRINSIC_INLINE bool rounds_to_nearest() noexcept {
1438 // https://lemire.me/blog/2020/06/26/gcc-not-nearest/
1439#if (FLT_EVAL_METHOD != 1) && (FLT_EVAL_METHOD != 0)
1440 return false;
1441#endif
1442 // See
1443 // A fast function to check your floating-point rounding mode
1444 // https://lemire.me/blog/2022/11/16/a-fast-function-to-check-your-floating-point-rounding-mode/
1445 //
1446 // This function is meant to be equivalent to :
1447 // prior: #include <cfenv>
1448 // return fegetround() == FE_TONEAREST;
1449 // However, it is expected to be much faster than the fegetround()
1450 // function call.
1451 //
1452 // The volatile keywoard prevents the compiler from computing the function
1453 // at compile-time.
1454 // There might be other ways to prevent compile-time optimizations (e.g.,
1455 // asm). The value does not need to be std::numeric_limits<float>::min(),
1456 // any small value so that 1 + x should round to 1 would do (after
1457 // accounting for excess precision, as in 387 instructions).
1458 static volatile float fmin = std::numeric_limits<float>::min();
1459 float fmini = fmin; // we copy it so that it gets loaded at most once.
1460//
1461// Explanation:
1462// Only when fegetround() == FE_TONEAREST do we have that
1463// fmin + 1.0f == 1.0f - fmin.
1464//
1465// FE_UPWARD:
1466// fmin + 1.0f > 1
1467// 1.0f - fmin == 1
1468//
1469// FE_DOWNWARD or FE_TOWARDZERO:
1470// fmin + 1.0f == 1
1471// 1.0f - fmin < 1
1472//
1473// Note: This may fail to be accurate if fast-math has been
1474// enabled, as rounding conventions may not apply.
1475#ifdef WJR_COMPILER_MSVC
1476 #pragma warning(push)
1477// todo: is there a VS warning?
1478// see
1479// https://stackoverflow.com/questions/46079446/is-there-a-warning-for-floating-point-equality-checking-in-visual-studio-2013
1480#elif defined(__clang__)
1481 #pragma clang diagnostic push
1482 #pragma clang diagnostic ignored "-Wfloat-equal"
1483#elif defined(__GNUC__)
1484 #pragma GCC diagnostic push
1485 #pragma GCC diagnostic ignored "-Wfloat-equal"
1486#endif
1487 return (fmini + 1.0f == 1.0f - fmini);
1488#ifdef WJR_COMPILER_MSVC
1489 #pragma warning(pop)
1490#elif defined(__clang__)
1491 #pragma clang diagnostic pop
1492#elif defined(__GNUC__)
1493 #pragma GCC diagnostic pop
1494#endif
1495}

1507template <typename Writer, typename Op>
1508from_chars_result<> __from_chars_impl(const char *first, const char *last, Writer wr,
1509 Op options) noexcept {
1510 static_assert(!std::is_reference_v<Writer>);
1511
1512 using T = typename Writer::float_type;
1513 constexpr bool is_support_integral = Writer::support_integral::value;
1514 constexpr bool is_constant_options = !std::is_same_v<Op, chars_format>;
1515
1516 if (WJR_UNLIKELY(first == last)) {
1517 return {first, std::errc::invalid_argument};
1518 }
1519
1520 const char *p = first;
1521 const auto fmt = to_underlying(static_cast<chars_format>(options));
1522
1523 parsed_number_string pns;
1524 pns.negative = (*p == '-');
1525
1526 if (*p == '-') { // C++17 20.19.3.(7.1) explicitly forbids '+' sign here
1527 if (++p == last) {
1528 return {first, std::errc::invalid_argument};
1529 }
1530 }
1531
1532 const char *const start_digits = p;
1533 uint64_t uval = 0; // an unsigned int avoids signed overflows (which are bad)
1534
1535 const char *end_of_integer_part;
1536 int64_t digit_count;
1537 int64_t exponent;
1538 int64_t exp_number; // explicit exponential part
1539
1540 do {
1541 uint32_t ch = *p - '0';
1542 if (ch >= 10) { // This situation rarely occurs
1543 if constexpr (is_constant_options) {
1544 // digit_count is zero, this is an error in json.
1545 if (fmt & to_underlying(chars_format::__json_format)) {
1546 return {first, std::errc::invalid_argument};
1547 }
1548 }
1549
1550 break;
1551 }
1552
1553 do {
1554 // a multiplication by 10 is cheaper than an arbitrary integer
1555 // multiplication
1556 uval = 10 * uval + ch; // might overflow, we will handle the overflow later
1557
1558 if (++p == last) {
1559 goto INTEGER_AT_END;
1560 }
1561
1562 ch = *p - '0';
1563 } while (WJR_LIKELY(ch < 10));
1564 } while (0);
1565
1566 do {
1567 end_of_integer_part = p;
1568 digit_count = static_cast<int64_t>(p - start_digits);
1569 pns.integer = span<const char>(start_digits, static_cast<size_t>(digit_count));
1570
1571 if constexpr (is_constant_options) {
1572 if (fmt & to_underlying(chars_format::__json_format)) {
1573 // at least 1 digit in integer part, without leading zeros
1574 if (digit_count == 0 || (start_digits[0] == '0' && digit_count > 1)) {
1575 return {first, std::errc::invalid_argument};
1576 }
1577 }
1578 }
1579
1580 if (*p != '.') {
1581 exponent = 0;
1582 if (*p == 'e' || *p == 'E') {
1583 break;
1584 }
1585
1586 goto INTEGER;
1587 }
1588
1589 ++p;
1590 const char *before = p;
1591 // can occur at most twice without overflowing, but let it occur more,
1592 // since for integers with many digits, digit parsing is the primary
1593 // bottleneck.
1594 if (last - p >= 8) {
1595 if (is_made_of_eight_digits_fast(p)) {
1596 do {
1597 uval = uval * 100000000 + parse_eight_digits_unrolled(p);
1598 p += 8;
1599
1600 if (last - p < 8) {
1601 while ((p != last) && is_integer(*p)) {
1602 uval = uval * 10 + uint32_t(*p - '0');
1603 ++p;
1604 }
1605
1606 goto POINT_DONE;
1607 }
1608
1609 } while (is_made_of_eight_digits_fast(p));
1610 }
1611
1612 while (is_integer(*p)) {
1613 uval = uval * 10 + uint32_t(*p - '0');
1614 ++p;
1615 }
1616 } else {
1617 while ((p != last) && is_integer(*p)) {
1618 uval = uval * 10 + uint32_t(*p - '0');
1619 ++p;
1620 }
1621 }
1622
1623 POINT_DONE:
1624
1625 exponent = before - p;
1626 pns.fraction = span<const char>(before, size_t(p - before));
1627 digit_count -= exponent;
1628
1629 auto &float_v = wr.get_float();
1630 if constexpr (is_constant_options) {
1631 if (fmt & to_underlying(chars_format::__json_format)) {
1632 if (WJR_UNLIKELY(exponent == 0)) {
1633 return detail::parse_infnan(first, last, float_v);
1634 }
1635 } else {
1636 if (WJR_UNLIKELY(digit_count == 0)) {
1637 return detail::parse_infnan(first, last, float_v);
1638 }
1639 }
1640 } else {
1641 if (WJR_UNLIKELY(digit_count == 0)) {
1642 return detail::parse_infnan(first, last, float_v);
1643 }
1644 }
1645 } while (0);
1646
1647 exp_number = 0;
1648
1649 if (bool(fmt & to_underlying(chars_format::scientific)) && (p != last) &&
1650 (('e' == *p) || ('E' == *p))) {
1651 const char *location_of_e = p;
1652 ++p;
1653 bool neg_exp = false;
1654 if ((p != last) && ('-' == *p)) {
1655 neg_exp = true;
1656 ++p;
1657 } else if ((p != last) &&
1658 ('+' == *p)) { // '+' on exponent is allowed by C++17 20.19.3.(7.1)
1659 ++p;
1660 }
1661 if ((p == last) || !is_integer(*p)) {
1662 if (!bool(fmt & to_underlying(chars_format::fixed))) {
1663 // We are in error.
1664 return detail::parse_infnan(first, last, wr.get_float());
1665 }
1666 // Otherwise, we will be ignoring the 'e'.
1667 p = location_of_e;
1668 } else {
1669 while ((p != last) && is_integer(*p)) {
1670 const auto digit = uint32_t(*p - '0');
1671 if (exp_number < 0x10000000) {
1672 exp_number = 10 * exp_number + digit;
1673 }
1674 ++p;
1675 }
1676 if (neg_exp) {
1677 exp_number = -exp_number;
1678 }
1679 exponent += exp_number;
1680 }
1681 } else {
1682 // If it scientific and not fixed, we have to bail out.
1683 if (bool(fmt & to_underlying(chars_format::scientific)) &&
1684 !bool(fmt & to_underlying(chars_format::fixed))) {
1685 return detail::parse_infnan(first, last, wr.get_float());
1686 }
1687 }
1688
1689 do {
1690 from_chars_result<> answer;
1691 answer.ptr = p;
1692 answer.ec = std::errc(); // be optimistic
1693 bool too_many_digits = false;
1694
1695 // If we frequently had to deal with long strings of digits,
1696 // we could extend our code by using a 128-bit integer instead
1697 // of a 64-bit integer. However, this is uncommon.
1698 //
1699 // We can deal with up to 19 digits.
1700 if (digit_count > 19) { // this is uncommon
1701 // It is possible that the integer had an overflow.
1702 // We have to handle the case where we have 0.0000somenumber.
1703 // We need to be mindful of the case where we only have zeroes...
1704 // E.g., 0.000000000...000.
1705 const char *start = start_digits;
1706 while ((start != last) && (*start == '0' || *start == '.')) {
1707 if (*start == '0') {
1708 digit_count--;
1709 }
1710 start++;
1711 }
1712
1713 if (digit_count > 19) {
1714 too_many_digits = true;
1715 // Let us start again, this time, avoiding overflows.
1716 // We don't need to check if is_integer, since we use the
1717 // pre-tokenized spans from above.
1718 uval = 0;
1719 p = pns.integer.data();
1720 const char *int_end = p + pns.integer.size();
1721 constexpr uint64_t minimal_nineteen_digit_integer = 1000000000000000000ull;
1722 while ((uval < minimal_nineteen_digit_integer) && (p != int_end)) {
1723 uval = uval * 10 + uint64_t(*p - '0');
1724 ++p;
1725 }
1726 if (uval >= minimal_nineteen_digit_integer) { // We have a big integers
1727 exponent = end_of_integer_part - p + exp_number;
1728 } else { // We have a value with a fractional component.
1729 p = pns.fraction.data();
1730 const char *frac_end = p + pns.fraction.size();
1731 while ((uval < minimal_nineteen_digit_integer) && (p != frac_end)) {
1732 uval = uval * 10 + uint64_t(*p - '0');
1733 ++p;
1734 }
1735 exponent = pns.fraction.data() - p + exp_number;
1736 }
1737 // We have now corrected both exponent and uval, to a truncated
1738 // value
1739 }
1740 }
1741
1742 pns.exponent = exponent;
1743
1744 T &float_v = wr.get_float();
1745
1746 // The implementation of the Clinger's fast path is convoluted because
1747 // we want round-to-nearest in all cases, irrespective of the rounding
1748 // mode selected on the thread. We proceed optimistically, assuming that
1749 // detail::rounds_to_nearest() returns true.
1750 if (binary_format<T>::min_exponent_fast_path() <= pns.exponent &&
1751 pns.exponent <= binary_format<T>::max_exponent_fast_path() && !too_many_digits) {
1752 // Unfortunately, the conventional Clinger's fast path is only
1753 // possible when the system rounds to the nearest float.
1754 //
1755 // We expect the next branch to almost always be selected.
1756 // We could check it first (before the previous branch), but
1757 // there might be performance advantages at having the check
1758 // be last.
1759 if (detail::rounds_to_nearest()) {
1760 // We have that fegetround() == FE_TONEAREST.
1761 // Next is Clinger's fast path.
1762 if (uval <= binary_format<T>::max_mantissa_fast_path()) {
1763 float_v = T(uval);
1764 if (pns.exponent < 0) {
1765 float_v = float_v / binary_format<T>::exact_power_of_ten(-pns.exponent);
1766 } else {
1767 float_v = float_v * binary_format<T>::exact_power_of_ten(pns.exponent);
1768 }
1769 if (pns.negative) {
1770 float_v = -float_v;
1771 }
1772 return answer;
1773 }
1774 } else {
1775 // We do not have that fegetround() == FE_TONEAREST.
1776 // Next is a modified Clinger's fast path, inspired by Jakub
1777 // Jelínek's proposal
1778 if (pns.exponent >= 0 &&
