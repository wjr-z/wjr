47template <size_t Size, size_t Min, size_t Max>
48WJR_INTRINSIC_INLINE void fallback_small_copy(const uint_t<Size> *first, const uint_t<Size> *last,
49 uint_t<Size> *dst) noexcept {
50 if constexpr (Min == 0) {
51 if (WJR_UNLIKELY(first == last)) {
52 return;
53 }
54 }
55
56 const auto n = to_unsigned(last - first);
57 WJR_ASSUME(n >= Min && n <= Max);
58
59 if constexpr (Max > 4) {
60 if (WJR_UNLIKELY(n > 4)) {
61 __scopy_detail::__small_copy<Size * 8>(first, last, dst, dst + n);
62 return;
63 }
64 }
65
66 if constexpr (Max > 2) {
67 if (WJR_LIKELY(n >= 2)) {
68 __scopy_detail::__small_copy<Size * 4>(first, last, dst, dst + n);
69 return;
70 }
71 }
72
73 dst[0] = first[0];
74}
