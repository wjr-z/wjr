36template <typename T>
37WJR_INTRINSIC_INLINE T div1by1_div_half(T hi, T lo, T den, T &rem) noexcept {
38 constexpr auto nd = std::numeric_limits<T>::digits;
39 using U = uint_t<nd * 2>;
40 const U n = (static_cast<U>(hi) << nd) | lo;
41 const auto result = static_cast<T>(n / den);
42 rem = n - result * static_cast<U>(den);
43 return result;
44}

54template <branch type, typename T>
55WJR_CONST WJR_INTRINSIC_INLINE div1by1_uint_t<T> div1by1_internal_uint_gen(T d) noexcept {
56 WJR_ASSERT(d != 0, "Can't divide by zero.");
57
58 constexpr auto nd = std::numeric_limits<T>::digits;
59
60 div1by1_uint_t<T> result;
61 const uint8_t floor_log_2_d = static_cast<uint8_t>(nd - 1 - clz<T>(d));
62
63 // Power of 2
64 if (is_zero_or_single_bit(d)) {
65 // We need to subtract 1 from the shift value in case of an unsigned
66 // branchfree div1by1_divider because there is a hardcoded right shift
67 // by 1 in its division algorithm. Because of this we also need to add
68 // back 1 in its recovery algorithm.
69 result.magic = 0;
70 result.more = static_cast<uint8_t>(floor_log_2_d - (type == branch::free));
71 } else {
72 uint8_t more;
73 T rem, proposed_m;
74 proposed_m = div1by1_div_half<T>(static_cast<T>(1) << floor_log_2_d, 0, d, rem);
75
76 WJR_ASSERT(rem > 0 && rem < d);
77 const T e = d - rem;
78
79 if (type == branch::full && (e < (static_cast<T>(1) << floor_log_2_d))) {
80 more = floor_log_2_d;
81 } else {
82 proposed_m += proposed_m;
83 const T twice_rem = rem + rem;
84 if (twice_rem >= d || twice_rem < rem) {
85 ++proposed_m;
86 }
87
88 more = floor_log_2_d | DIV1BY1_ADD_MARKER;
89 }
90
91 result.magic = 1 + proposed_m;
92 result.more = more;
93 }
94
95 return result;
96}

115template <typename T, branch type>
116WJR_CONST WJR_INTRINSIC_INLINE div1by1_uint_t<T> div1by1_uint_gen(T d) noexcept {
117 if constexpr (type == branch::full) {
118 return div1by1_internal_uint_gen<type>(d);
119 } else {
120 WJR_ASSERT(d != 1, "The divisor cannot be 1");
121 const auto ret = div1by1_internal_uint_gen<type>(d);
122 return div1by1_uint_t<T>{ret.magic,
123 static_cast<uint8_t>(ret.more & div1by1_shift_mask<T>::value)};
124 }

127template <typename T>
128WJR_CONST WJR_INTRINSIC_INLINE T
129div1by1_uint_branchfull_do_impl(T d, const div1by1_uint_t<T> &denom) noexcept {
130 const T magic = denom.magic;
131 const uint8_t more = denom.more;
132 if (!magic) {
133 return d >> more;
134 } else {
135 const T q = mulhi<T>(magic, d);
136 if (more & DIV1BY1_ADD_MARKER) {
137 const T t = ((d - q) >> 1) + q;
138 return t >> (more & div1by1_shift_mask<T>::value);
139 } else {
140 return q >> more;
141 }
142 }
143}
