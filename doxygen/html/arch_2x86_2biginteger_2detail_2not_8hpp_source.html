29template <typename T>
30void large_builtin_bi_not_n(T *dst, const T *src, size_t n) noexcept {
31 constexpr auto is_avx = WJR_HAS_SIMD(AVX2);
32
33 using simd = std::conditional_t<is_avx, avx, sse>;
34 constexpr auto simd_width = simd::width();
35 constexpr auto type_width = simd_width / 64;
36
37 const uintptr_t ptr = reinterpret_cast<uintptr_t>(dst);
38 WJR_ASSUME(ptr % sizeof(T) == 0);
39 const size_t offset = align_up_offset(ptr, 32) / sizeof(T);
40
41 WJR_ASSUME(offset < 4);
42
43 const auto y = sse::ones();
44
45 do {
46 if (offset == 0) {
47 break;
48 }
49
50 if (offset == 1) {
51 dst[0] = ~src[0];
52 break;
53 }
54
55 sse::storeu(dst, sse::Xor(sse::loadu(src), y));
56
57 if (offset == 3) {
58 dst[2] = ~src[2];
59 }
60 } while (0);
61
62 dst += offset;
63 src += offset;
64 n -= offset;
65
66 const auto z = broadcast<__m128i_t, typename simd::int_tag_type>(y);
67
68 size_t m = n & (-type_width * 4);
69 n &= (type_width * 4) - 1;
70 WJR_ASSUME(m != 0);
71
72 do {
73 auto x0 = simd::loadu(src);
74 auto x1 = simd::loadu(src + type_width);
75 auto x2 = simd::loadu(src + type_width * 2);
76 auto x3 = simd::loadu(src + type_width * 3);
77
78 simd::store(dst, simd::Xor(x0, z));
79 simd::store(dst + type_width, simd::Xor(x1, z));
80 simd::store(dst + type_width * 2, simd::Xor(x2, z));
81 simd::store(dst + type_width * 3, simd::Xor(x3, z));
82
83 dst += type_width * 4;
84 src += type_width * 4;
85 m -= type_width * 4;
86 } while (m != 0);
87
88 if (WJR_UNLIKELY(n == 0)) {
89 return;
90 }
91
92 m = n / type_width;
93 WJR_ASSUME(m < 4);
94
95 switch (m) {
96 case 3: {
97 simd::store(dst, simd::Xor(simd::loadu(src), z));
98 WJR_FALLTHROUGH;
99 }
100 case 2: {
101 simd::store(dst + type_width * (m - 2),
102 simd::Xor(simd::loadu(src + type_width * (m - 2)), z));
103 WJR_FALLTHROUGH;
104 }
105 case 1: {
106 simd::store(dst + type_width * (m - 1),
107 simd::Xor(simd::loadu(src + type_width * (m - 1)), z));
108 WJR_FALLTHROUGH;
109 }
110 case 0: {
111 break;
112 }
113 default: {
114 WJR_UNREACHABLE();
115 }
116 }
117
118 m = n & (-type_width);
119
120 if (WJR_UNLIKELY(n == m)) {
121 return;
122 }
123
124 WJR_ASSUME(n - m < 4);
125
126 switch (n - m) {
127 case 1: {
128 dst[m] = ~src[m];
129 break;
130 }
131 case 2: {
132 sse::store(dst + m, sse::Xor(sse::loadu(src + m), y));
133 break;
134 }
135
136 case 3: {
137 sse::store(dst + m, sse::Xor(sse::loadu(src + m), y));
138 dst[m + 2] = ~src[m + 2];
139 break;
140 }
141
142 default: {
143 WJR_UNREACHABLE();
144 }
145 }
146
147 return;
148}

150template <typename T>
151WJR_INTRINSIC_INLINE void unaligned_large_builtin_bi_not_n(T *dst, const T *src,
152 size_t n) noexcept {
153 size_t idx = 0;
154 const auto y = sse::ones();
155
156 if (n & 4) {
157 sse::storeu(dst + idx, sse::Xor(sse::loadu(src + idx), y));
158 sse::storeu(dst + idx + 2, sse::Xor(sse::loadu(src + idx + 2), y));
159
160 idx += 4;
161 }
162
163 if (n & 2) {
164 sse::storeu(dst + idx, sse::Xor(sse::loadu(src + idx), y));
165
166 idx += 2;
167 }
168
169 if (n & 1) {
170 dst[idx] = ~src[idx];
171
172 ++idx;
173 }
174
175 if (WJR_UNLIKELY(idx == n)) {
176 return;
177 }
178
179 WJR_ASSUME((n - idx) % 8 == 0);
180
181 do {
182 auto x0 = sse::loadu(src + idx);
183 auto x1 = sse::loadu(src + idx + 2);
184 auto x2 = sse::loadu(src + idx + 4);
185 auto x3 = sse::loadu(src + idx + 6);
186
187 sse::storeu(dst + idx, sse::Xor(x0, y));
188 sse::storeu(dst + idx + 2, sse::Xor(x1, y));
189 sse::storeu(dst + idx + 4, sse::Xor(x2, y));
190 sse::storeu(dst + idx + 6, sse::Xor(x3, y));
191
192 idx += 8;
193 } while (idx != n);
194}

196template <typename T>
197WJR_INTRINSIC_INLINE void builtin_bi_not_n(T *dst, const T *src, size_t n) noexcept {
198 static_assert(sizeof(T) == 8);
199
200 if (n < 4) {
201 if (WJR_UNLIKELY(n == 0)) {
202 return;
203 }
204
205 if (WJR_UNLIKELY(n == 1)) {
206 dst[0] = ~src[0];
207 return;
208 }
209
210 sse::storeu(dst, sse::Xor(sse::loadu(src), sse::ones()));
211
212 if (n == 3) {
213 dst[2] = ~src[2];
214 }
215
216 return;
217 }
218
219 // Can be aligned
220 // todo : Align those that cannot be aligned with T through uint8_t
221 if (WJR_LIKELY(n >= 35 && reinterpret_cast<uintptr_t>(dst) % sizeof(T) == 0)) {
222 return large_builtin_bi_not_n(dst, src, n);
223 }
224
225 return unaligned_large_builtin_bi_not_n(dst, src, n);
226}
