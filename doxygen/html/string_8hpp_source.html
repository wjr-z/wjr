192template <unsigned int Length, WJR_REQUIRES(Length <= 8)>
193WJR_PURE WJR_INTRINSIC_INLINE bool
194constant_length_strncaseequal(const char *a, const char *b,
195 integral_constant<unsigned int, Length>) {
196 if constexpr (Length == 1) {
197 const char diff = (a[0] ^ b[0]) & 0xDF;
198 return diff == 0;
199 } else if constexpr (Length == 2 || Length == 4 || Length == 8) {
200 using diff_type = uint_t<Length * 8>;
201 constexpr diff_type mask = broadcast<uint8_t, diff_type>(0xDF);
202 diff_type diff = (read_memory<diff_type>(a) ^ read_memory<diff_type>(b)) & mask;
203 return diff == 0;
204 } else if constexpr (Length == 3) {
205 return constant_length_strncaseequal(a, b, 2_u) &
206 constant_length_strncaseequal(a + 2, b + 2, 1_u);
207 } else if constexpr (Length >= 5 && Length <= 6) {
208 constexpr auto RestLength = Length - 4;
209 return constant_length_strncaseequal(a, b, 4_u) &
210 constant_length_strncaseequal(a + 4, b + 4,
211 integral_constant<unsigned int, RestLength>());
212 } else {
213 static_assert(Length == 7);
214 return constant_length_strncaseequal(a, b, 4_u) &
215 constant_length_strncaseequal(a + 3, b + 3, 4_u);
216 }
217}

219WJR_PURE constexpr int strncasecmp(const char *a, const char *b, size_t length) {
220 for (size_t i = 0; i < length; ++i) {
221 const char diff = (a[i] ^ b[i]) & 0xDF;
222 if (diff != 0) {
223 return a[i] < b[i] ? -1 : 1;
224 }
225 }
226
227 return 0;
228}

230WJR_PURE constexpr int compare_insensitive(std::string_view lhs, std::string_view rhs) {
231 if (int ret = strncasecmp(lhs.data(), rhs.data(), std::min(lhs.length(), rhs.length()))) {
232 return ret;
233 }
234
235 return lhs.size() == rhs.size() ? 0 : (lhs.size() < rhs.size() ? -1 : 1);
236}
