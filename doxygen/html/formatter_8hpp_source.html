67template <typename Container>
68WJR_INTRINSIC_INLINE void format_string(Container &cont, std::string_view str) {
69 auto first = str.data();
70 const auto length = str.size();
71
72 const auto old_size = cont.size();
73 try_uninitialized_append(cont, length + 2);
74 auto *ptr = cont.data() + old_size;
75 *ptr++ = '\"';
76
77 size_t pos = 0;
78
79 for (; pos + 8 <= length; pos += 8) {
80 // Poor's man vectorization. This could get much faster if we used SIMD.
81 //
82 // It is not the case that replacing '|' with '||' would be neutral
83 // performance-wise.
84 if (needs_escaping[uint8_t(first[pos])] | needs_escaping[uint8_t(first[pos + 1])] |
85 needs_escaping[uint8_t(first[pos + 2])] | needs_escaping[uint8_t(first[pos + 3])] |
86 needs_escaping[uint8_t(first[pos + 4])] | needs_escaping[uint8_t(first[pos + 5])] |
87 needs_escaping[uint8_t(first[pos + 6])] | needs_escaping[uint8_t(first[pos + 7])]) {
88 goto SLOW_PATH;
89 }
90 }
91
92 for (; pos < length; pos++) {
93 if (needs_escaping[uint8_t(first[pos])]) {
94 goto SMALL_SLOW_PATH;
95 }
96 }
97
98 // The following is also possible and omits a 256-byte table, but it is
99 // slower: for (; (pos < unescaped.length()) && (uint8_t(unescaped[pos]) >
100 // 0x1F)
101 // && (unescaped[pos] != '\"') && (unescaped[pos] != '\\'); pos++) {}
102
103 // At least for long strings, the following should be fast. We could
104 // do better by integrating the checks and the insertion.
105 std::memcpy(ptr, first, length);
106 ptr += length;
107 *ptr++ = '\"';
108 return;
109
110SLOW_PATH: {
111 std::memcpy(ptr, first, pos);
112 try_uninitialized_resize(cont, (ptr + pos) - cont.data());
113
114 for (int i = 0; i < 8; ++i) {
115 switch (first[pos + i]) {
116 case '\"': {
117 append_string(cont, "\\\"", 2);
118 break;
119 }
120 case '\\': {
121 append_string(cont, "\\\\", 2);
122 break;
123 }
124 default:
125 if (uint8_t(first[pos + i]) <= 0x1F) {
126 const auto u = escaped[uint8_t(first[pos + i])];
127 if (u.length == 2) {
128 append_string(cont, u.string, 2);
129 } else {
130 append_string(cont, u.string, 6);
131 }
132 } else {
133 cont.push_back(first[pos + i]);
134 }
135 } // switch
136 }
137
138 pos += 8;
139
140 if (pos + 8 <= length) {
141 size_t last_pos = pos;
142
143 do {
144 if (needs_escaping[uint8_t(first[pos])] | needs_escaping[uint8_t(first[pos + 1])] |
145 needs_escaping[uint8_t(first[pos + 2])] | needs_escaping[uint8_t(first[pos + 3])] |
146 needs_escaping[uint8_t(first[pos + 4])] | needs_escaping[uint8_t(first[pos + 5])] |
147 needs_escaping[uint8_t(first[pos + 6])] | needs_escaping[uint8_t(first[pos + 7])]) {
148 append_string(cont, first + last_pos, pos - last_pos);
149 last_pos = pos + 8;
150
151 for (int i = 0; i < 8; ++i) {
152 switch (first[pos + i]) {
153 case '\"': {
154 append_string(cont, "\\\"", 2);
155 break;
156 }
157 case '\\': {
158 append_string(cont, "\\\\", 2);
159 break;
160 }
161 default:
162 if (uint8_t(first[pos + i]) <= 0x1F) {
163 const auto u = escaped[uint8_t(first[pos + i])];
164 if (u.length == 2) {
165 append_string(cont, u.string, 2);
166 } else {
167 append_string(cont, u.string, 6);
168 }
169 } else {
170 cont.push_back(first[pos + i]);
171 }
172 } // switch
173 }
174 }
175
176 pos += 8;
177 } while (pos + 8 <= length);
178
179 append_string(cont, first + last_pos, pos - last_pos);
180 }
181
182 do {
183 break;
184
185 SMALL_SLOW_PATH:
186 std::memcpy(ptr, first, pos);
187 try_uninitialized_resize(cont, (ptr + pos) - cont.data());
188
189 } while (0);
190
191 // We caught a control character if we enter this loop (slow).
192 // Note that we are do not restart from the beginning, but rather we
193 // continue from the point where we encountered something that requires
194 // escaping.
195 for (; pos < length; pos++) {
196 switch (first[pos]) {
197 case '\"': {
198 append_string(cont, "\\\"", 2);
199 break;
200 }
201 case '\\': {
202 append_string(cont, "\\\\", 2);
203 break;
204 }
205 default:
206 if (uint8_t(first[pos]) <= 0x1F) {
207 const auto u = escaped[uint8_t(first[pos])];
208 if (u.length == 2) {
209 append_string(cont, u.string, 2);
210 } else {
211 append_string(cont, u.string, 6);
212 }
213 } else {
214 cont.push_back(first[pos]);
215 }
216 } // switch
217 } // for
218
219 cont.push_back('\"');
220 return;
221}
