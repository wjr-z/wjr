15 const div2by1_divider_noshift<uint64_t> &div) noexcept {

68 WJR_ASSERT_ASSUME(n >= 1);
69
70 if (WJR_UNLIKELY(div.is_zero_or_single_bit())) {

87WJR_INTRINSIC_INLINE void div_qr_1(uint64_t *dst, uint64_t &rem, const uint64_t *src, size_t n,
88 uint64_t div) noexcept {
89 WJR_ASSERT_ASSUME(n >= 1);
90 WJR_ASSERT_ASSUME(div != 0);
91
92 if (WJR_UNLIKELY(is_zero_or_single_bit(div))) {
93 const unsigned int c = ctz(div);
94 rem = src[0] & ((1ull << c) - 1);
95 (void)rshift_n(dst, src, n, c);
96 return;
97 }
98
99 if (WJR_UNLIKELY(n == 1)) {
100 const uint64_t tmp = src[0];
101
102 if (__has_high_bit(div)) {
103 if (tmp >= div) {
104 rem = tmp - div;
105 dst[0] = 1;
106 return;
107 }
108 rem = tmp;
109 dst[0] = 0;
110 return;
111 }
112

115 return;

120 dst[0] = ax;
121 dst[1] = dx;
122 return;
123 }
124

239void fallback_divexact_1_noshift(uint64_t *dst, const uint64_t *src, size_t n,
240 const divexact1_divider<uint64_t> &div) noexcept {
241 const uint64_t divisor = div.get_divisor();
242 const uint64_t value = div.get_value();
243
244 uint64_t rdx = 0, r10 = 0;
245 uint64_t cf = 0;
246 size_t idx = 0;
247
248 --n;
249
250 if (WJR_LIKELY(n != 0)) {
251 do {
252 r10 = src[idx];
253 r10 = subc(r10, rdx, cf, cf);
254 r10 = mullo(r10, value);
255 dst[idx] = r10;
256 ++idx;
257 rdx = mulhi(r10, divisor);
258 } while (WJR_LIKELY(idx != n));
259 }
260
261 r10 = src[n];
262 r10 -= rdx + cf;
263 r10 = mullo(r10, value);
264 dst[n] = r10;
265 return;
266}

268WJR_INLINE_CONSTEXPR20 void fallback_divexact_1_shift(uint64_t *dst, const uint64_t *src, size_t n,
269 const divexact1_divider<uint64_t> &div,
270 uint64_t hicf = 0) noexcept {
271 const uint64_t divisor = div.get_divisor();
272 const uint64_t value = div.get_value();
273 const auto shift = div.get_shift();
274
275 uint64_t rdx = 0, r10 = 0;
276 uint64_t cf = 0;
277 size_t idx = 0;
278
279 --n;
280 r10 = src[0];
281
282 if (WJR_LIKELY(n != 0)) {
283 do {
284 uint64_t r11 = src[idx + 1];
285 r10 = shrd(r10, r11, shift);
286 r10 = subc(r10, rdx, cf, cf);
287 r10 = mullo(r10, value);
288 dst[idx] = r10;
289 ++idx;
290 rdx = mulhi(r10, divisor);
291 r10 = r11;
292 } while (WJR_LIKELY(idx != n));
293 }
294
295 r10 = shrd(r10, hicf, shift);
296 r10 -= rdx + cf;
297 r10 = mullo(r10, value);
298 dst[n] = r10;
299 return;

328template <uint64_t c>
329constexpr __divexact_get_struct __divexact_init() noexcept {
330 if constexpr (is_zero_or_single_bit(c)) {
331 return {0, constant::ctz(c), 0, 0};
332 } else {
333 constexpr auto p0 = __divexact_get<c>();
334 if constexpr (p0 == 1) {
335 return {1, 0, 0, 0};
336 } else {
337 constexpr auto c0 = c / p0;
338 if constexpr (is_zero_or_single_bit(c0)) {
339 return {2, constant::ctz(c), p0, 0};
340 } else {
341 constexpr auto p1 = __divexact_get<c0>();
342 if constexpr (p1 == 1) {
343 return {1, 0, 0, 0};
344 } else {
345 constexpr auto c1 = c0 / p1;
346 if constexpr (is_zero_or_single_bit(c1)) {
347 return {3, constant::ctz(c1), p0, p1};
348 } else {
349 return {1, 0, 0, 0};
350 }
351 }

357template <uint64_t c>
358WJR_INTRINSIC_CONSTEXPR20 void divexact_byc(uint64_t *dst, const uint64_t *src, size_t n,
359 integral_constant<uint64_t, c>,
360 WJR_MAYBE_UNUSED uint64_t cf) noexcept {
361 // cost : divexact_dbm1c * 2 + shift * 1 <= divexact_1
362 static_assert(c != 0);
363 constexpr auto ss = __divexact_init<c>();
364
365 if constexpr (ss.mode == 0) {
366 (void)rshift_n(dst, src, n, ss.cl, cf);
367 }
368
369 if constexpr (ss.mode == 1) {
370 constexpr auto shift = constant::ctz(c);
371 using divider_t = divexact1_divider<uint64_t>;
372 constexpr auto divisor = c >> shift;
373 constexpr auto value = divider_t::reciprocal(divisor);
374 constexpr auto divider = divider_t(divisor, value, shift);
375
376 static_assert(!divider.is_zero_or_single_bit());
377
378 if constexpr (shift == 0) {
379 fallback_divexact_1_noshift(dst, src, n, divider);
380 } else {
381 fallback_divexact_1_shift(dst, src, n, divider, cf);
382 }
383 }
384
385 if constexpr (ss.mode == 2 || ss.mode == 3) {
386 constexpr uint64_t maxn = UINT64_MAX;
387
388 if constexpr (ss.cl == 0) {
389 (void)divexact_dbm1c(dst, src, n, maxn / ss.p0, 0);
390 } else {
391 (void)rshift_n(dst, src, n, ss.cl, cf);
392 (void)divexact_dbm1c(dst, dst, n, maxn / ss.p0, 0);
393 }
394
395 if constexpr (ss.mode == 3) {
396 (void)divexact_dbm1c(dst, dst, n, maxn / ss.p1, 0);
397 }
398 }

471WJR_PURE WJR_INLINE_CONSTEXPR20 uint64_t
472mod_1_shift(const uint64_t *src, size_t n, const div2by1_divider<uint64_t> &div) noexcept {
473 WJR_ASSERT_ASSUME(n >= 1);
474 WJR_ASSERT(div.get_shift() != 0);
475
476 const uint64_t divisor = div.get_divisor();
477 const uint64_t value = div.get_value();
478 const auto shift = div.get_shift();
479
480 uint64_t lo, hi;
481
482 uint64_t rbp = src[n - 1];
483 --n;
484 hi = rbp >> (64 - shift);
485
486 do {
487 if (WJR_UNLIKELY(n == 0)) {
488 (void)div.divide(divisor, value, rbp << shift, hi);
489 break;
490 }
491
492 lo = src[n - 1];
493 (void)div.divide(divisor, value, shld(rbp, lo, shift), hi);
494 rbp = lo;
495 --n;
496
497 if (WJR_LIKELY(n != 0)) {
498 do {
499 lo = src[n - 1];
500 (void)div.divide(divisor, value, shld(rbp, lo, shift), hi);
501 rbp = lo;
502 --n;
503 } while (WJR_LIKELY(n != 0));
504 }
505
506 (void)div.divide(divisor, value, rbp << shift, hi);
507 } while (0);
508
509 return hi >> shift;
510}
