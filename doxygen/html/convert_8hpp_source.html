87template <typename Converter>
88uint8_t *basecase_to_chars(uint8_t *first, size_t len, uint64_t *up, size_t n, unsigned int base,
89 Converter conv) noexcept {
90 constexpr size_t buf_len = dc_bignum_to_chars_precompute_threshold * 64 * 7 / 11;
91 uint8_t buf[buf_len];
92 uint8_t *const end = buf + buf_len;
93 uint8_t *start;
94
95 if (WJR_LIKELY(base == 10)) {
96 start = basecase_to_chars_10(end, up, n, conv);
97 } else {
98 start = end;
99 }
100
101 if (const size_t rlen = end - start; len > rlen) {
102 first = std::fill_n(first, len - rlen, conv.template to<1>(0));
103 }
104
105 return copy_restrict(start, end, first);
106}

108template <typename Converter>
109uint8_t *dc_to_chars(uint8_t *first, size_t len, uint64_t *up, size_t n,
110 precompute_chars_convert_t *pre, uint64_t *stk, Converter conv) noexcept {
111 WJR_ASSERT_ASSUME(n >= 1);
112 if (n < dc_bignum_to_chars_threshold) {
113 return basecase_to_chars(first, len, up, n, pre->base, conv);
114 }
115
116 const auto *const pp = pre->ptr;
117 const size_t pn = pre->size;
118 const size_t ps = pre->shift;
119
120 WJR_ASSERT((pn + ps) * 5 >= n * 2);
121
122 if (n < pn + ps || (n == pn + ps && reverse_compare_n(up + ps, pp, pn) < 0)) {
123 return dc_to_chars(first, len, up, n, pre - 1, stk, conv);
124 }
125
126 const size_t pd = pre->digits_in_base;
127 auto *qp = stk;
128
129 div_qr_s(qp, up + ps, up + ps, n - ps, pp, pn);
130
131 size_t qn = n - pn - ps;
132 qn += qp[qn] != 0;
133
134 if (len != 0) {
135 len = len - pd;
136 }
137
138 pre -= qn * 2 <= n;
139
140 first = dc_to_chars(first, len, qp, qn, pre, stk + qn, conv);
141 first = dc_to_chars(first, pd, up, pn + ps, pre, stk, conv);
142 return first;
143}

198 const auto __size = NAME(__ptr, WJR_PP_QUEUE_EXPAND(CALL), conv) TAIL; \
199 WJR_ASSERT((size_t)__size <= SIZE); \
200 if constexpr (__fast_container_inserter_v == 1) { \
201 resize(cont, __presize + __size); \
202 } else { \

210 const auto __size = NAME(__ptr, WJR_PP_QUEUE_EXPAND(CALL), conv) TAIL; \

216 case 2: {

285template <typename Converter>
286size_t __biginteger_from_chars_2_impl(const uint8_t *first, size_t n, uint64_t *up,
287 Converter conv) noexcept {
288 const size_t hbits = (n - 1) % 64 + 1;
289 const size_t len = (n - 1) / 64 + 1;
290
291 uint64_t x = 0;
292 up += len;
293
294 __unsigned_from_chars_unchecked<2>(first, first + hbits, x, conv);
295 first += hbits;
296
297 *--up = x;
298
299 size_t idx = len - 1;
300
301 if (idx) {
302 do {
303 x = 0;
304
305 for (int i = 0; i < 4; ++i) {
306 x = (x << 16) + __from_chars_unroll_16<2>(first, conv);
307 first += 16;
308 }
309
310 *--up = x;
311 } while (WJR_LIKELY(--idx));
312 }
313
314 return len;
315}

329template <typename Converter>
330size_t __biginteger_from_chars_16_impl(const uint8_t *first, size_t n, uint64_t *up,
331 Converter conv) noexcept {
332 const size_t hbits = (n - 1) % 16 + 1;
333 const size_t len = (n - 1) / 16 + 1;
334
335 uint64_t x = 0;
336 up += len;
337
338 __unsigned_from_chars_unchecked<16>(first, first + hbits, x, conv);
339 first += hbits;
340
341 *--up = x;
342
343 size_t idx = len - 1;
344
345 if (idx) {
346 do {
347 x = 0;
348
349 for (int i = 0; i < 4; ++i) {
350 auto x0 = conv.template from<16>(first[0]);
351 auto x1 = conv.template from<16>(first[1]);
352 auto x2 = conv.template from<16>(first[2]);
353 auto x3 = conv.template from<16>(first[3]);
354
355 x = x << 16 | x0 << 12 | x1 << 8 | x2 << 4 | x3;
356 first += 4;
357 }
358
359 *--up = x;
360 } while (WJR_LIKELY(--idx));
361 }
362
363 return len;
364}

402uint64_t *__basecase_basecase_from_chars(const uint8_t *first, size_t n, uint64_t *up,
