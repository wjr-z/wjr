15WJR_INTRINSIC_CONSTEXPR T fallback_addc(T a, T b, U c_in, U &c_out) noexcept {
16 T ret = a;
17 ret += b;
18 U c = ret < b;
19 ret += c_in;
20 c |= ret < c_in;
21 c_out = c;
22 return ret;
23}

75template <typename T, typename U,
76 WJR_REQUIRES_I(is_nonbool_unsigned_integral_v<T> &&is_unsigned_integral_v<U>)>
77WJR_INTRINSIC_CONSTEXPR20 T addc(T a, T b, type_identity_t<U> c_in, U &c_out) noexcept {
78 WJR_ASSERT_ASSUME_L2(c_in <= 1);
79
80#if !WJR_HAS_BUILTIN(ADDC) && !WJR_HAS_BUILTIN(ASM_ADDC)
81 return fallback_addc(a, b, c_in, c_out);
82#else
83 constexpr auto is_constant_or_zero = [](auto x) -> int {
84 return WJR_BUILTIN_CONSTANT_P_TRUE(x == 0) ? 2 : WJR_BUILTIN_CONSTANT_P(x) ? 1 : 0;
85 };
86
87 // The compiler should be able to optimize the judgment condition of if when
88 // enabling optimization. If it doesn't work, then there should be a issue
89 if (is_constant_evaluated() ||
90 // constant value is zero or constant value number greater or equal than
91 // 2
92 (is_constant_or_zero(a) + is_constant_or_zero(b) + is_constant_or_zero(c_in) >= 2)) {
93 return fallback_addc(a, b, c_in, c_out);
94 }
95
96 if constexpr (sizeof(T) == 8) {
97 return WJR_PP_BOOL_IF_NE(
98 WJR_HAS_BUILTIN(ASM_ADDC), asm_addc,
99 WJR_PP_BOOL_IF(WJR_HAS_BUILTIN(ADDC), builtin_addc, fallback_addc))(a, b, c_in, c_out);
100 } else {
101 return WJR_PP_BOOL_IF(WJR_HAS_BUILTIN(ADDC), builtin_addc, fallback_addc)(a, b, c_in,
102 c_out);
103 }
104#endif
105}
