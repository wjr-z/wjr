38WJR_CONST WJR_INTRINSIC_INLINE __m128i builtin_to_chars_unroll_8_fast_10(uint32_t in) noexcept {
39 __m128i x = simd_cast<uint32_t, __m128i_t>(in);
40 __m128i q, r;
41
42 q = _mm_mul_epu32(x, to_chars_detail::mul10p4);
43 q = _mm_srli_epi64(q, 45);
44
45 r = _mm_sub_epi32(x, _mm_mul_epu32(q, to_chars_detail::mul10p4x));
46 x = _mm_packus_epi32(q, r);
47
48 q = _mm_mulhi_epu16(x, to_chars_detail::mul10p2);
49 q = _mm_srli_epi16(q, 3);
50
51 r = _mm_sub_epi16(x, _mm_mullo_epi16(q, to_chars_detail::mul10p2x));
52 x = _mm_packus_epi16(q, r);
53
54 q = _mm_mulhi_epu16(x, to_chars_detail::mul10p1);
55 r = _mm_sub_epi16(x, _mm_and_si128(q, to_chars_detail::mul10q8mask));
56 q = _mm_srli_epi16(q, 3);
57 r = _mm_sub_epi16(r, _mm_slli_epi16(q, 1));
58 x = _mm_packus_epi16(q, r);
59
60 return sse::shuffle_epi8(x, to_chars_detail::shuf);
61}
