103template <typename Getter>
104decltype(auto) build_tree(size_t n, Getter getter) noexcept {
105 using value_type = typename Getter::value_type;
106 static_assert(std::is_trivially_copyable_v<value_type>);
107 using tree_type = huffman_tree<value_type>;
108 using pair_type = __trivial_pair<value_type>;
109 using inner_type = huffman_node<value_type>;
110 using leaf_type = huffman_leaf_node<value_type>;
111
112 std::unique_ptr<char[]> buffer(new char[sizeof(size_t) * (n - 1) + sizeof(pair_type) * n]);
113
114 auto *size_first = reinterpret_cast<size_t *>(buffer.get()) + (n - 1);
115 auto *size_last = size_first;
116 auto *pair_first = reinterpret_cast<pair_type *>(size_first);
117
118 for (size_t i = 0; i < n; ++i) {
119 pair_first[i].first = getter.get_size(i);
120 pair_first[i].second = getter.get_value(i);
121 }
122
123 std::sort(pair_first, pair_first + n,
124 [](const auto &lhs, const auto &rhs) { return lhs.first < rhs.first; });
125
126 huffman_tree<value_type> tree((n - 1) * sizeof(inner_type) + n * sizeof(leaf_type));
127 inner_type *const inner_end = tree.root();
128 auto *inner_first = inner_end + (n - 1);
129 auto *inner_last = inner_first;
130 auto *leaf_first = reinterpret_cast<leaf_type *>(inner_first);
131 auto *const leaf_last = leaf_first + n;
132
133 for (size_t i = 0; i < n; ++i) {
134 wjr::construct_at(leaf_first + i, pair_first[i].second);
135 }
136
137 if (n != 1) {
138 do {
139 size_t &total_size = size_last[-1];
140 inner_type &new_node = inner_last[-1];
141 do {
142 if (inner_first == inner_last) {
143 total_size = pair_first[0].first + pair_first[1].first;
144 new_node.set_left(leaf_first);
145 new_node.set_left(leaf_first + 1);
146 leaf_first += 2;
147 pair_first += 2;
148 break;
149 }
150
151 size_t pair_size = pair_first->first;
152 do {
153 if (size_first[-1] < pair_size) {
154 total_size = size_first[-1];
155 --inner_first;
156 --size_first;
157 new_node.set_left(inner_first);
158
159 if (inner_first != inner_last && size_first[-1] < pair_size) {
160 total_size += size_first[-1];
161 --inner_first;
162 --size_first;
163 new_node.set_right(inner_first);
164
165 WJR_ASSUME(leaf_first != leaf_last);
166 break;
167 }
168
169 total_size += pair_size;
170 new_node.set_right(leaf_first);
171 ++leaf_first;
172 ++pair_first;
173 break;
174 }
175
176 total_size = pair_size;
177 new_node.set_left(leaf_first);
178 ++leaf_first;
179 ++pair_first;
180
181 if (leaf_first == leaf_last ||
182 (pair_size = pair_first->first, size_first[-1] < pair_size)) {
183 total_size += size_first[-1];
184 --inner_first;
185 --size_first;
186 new_node.set_right(inner_first);
187 break;
188 }
189
190 total_size += pair_size;
191 new_node.set_right(leaf_first);
192 ++leaf_first;
193 ++pair_first;
194 } while (0);
195 } while (0);
196
197 --inner_last;
198 --size_last;
199 } while (leaf_first != leaf_last);
200
201 if (inner_last != inner_end) {
202 do {
203 inner_type &new_node = inner_last[-1];
204 size_t &total_size = size_last[-1];
205
206 total_size = size_first[-1] + size_first[-2];
207 new_node.set_left(inner_first - 1);
208 new_node.set_right(inner_first - 2);
209 inner_first -= 2;
210 size_first -= 2;
211
212 --inner_last;
213 --size_last;
214 } while (inner_last != inner_end);
215 }
216 }
217
218 std::destroy_n(pair_first, n);
219 return tree;
220}
