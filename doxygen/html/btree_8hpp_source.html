52WJR_INTRINSIC_INLINE void copy(const T *first, const T *last, T *dest) noexcept {
53 if constexpr (Max <= 2) {
54 static_assert(Max >= 1);
55
56 if constexpr (Min == 0) {
57 if (first == last) {
58 return;
59 }
60 }
61
62 dest[0] = first[0];
63
64 if constexpr (Max == 2) {
65 if constexpr (Min <= 1) {
66 if (first + 1 == last) {
67 return;
68 }
69 }
70
71 dest[1] = first[1];
72 }
73 } else {
74 small_copy<Min, Max>(first, last, dest);
75 }

631 WJR_INTRINSIC_INLINE static unsigned int search_1_impl(unsigned int size,
632 const Compare &comp) noexcept {
633 unsigned int i = 1;
634 while (WJR_LIKELY(i < size)) {
635 if (comp(i)) {
636 // prop : 50%
637 return i - (int)comp(i - 1);

1296 unsigned int size = __get_size();

1298 __drop_node(__get_base()->m_values[i]);
1299 }
1300 return;
1301 }

1303 int cur_size = current->size();
1304
1305 // If root is leaf
1306 if (cur_size < 0) {
1307 auto *const leaf = current->as_leaf();
1308 const unsigned int cur_usize = -cur_size;
1309

1311 __drop_node(leaf->m_values[i]);

1320 cur_size = -current->size();
1321
