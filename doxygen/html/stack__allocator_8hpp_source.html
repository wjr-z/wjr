56 static void *__large_allocate(size_t n, large_memory *&mem) noexcept {
57 auto *const raw = mem::__default_allocate<>(n + aligned_header_size, std::nothrow);
58 auto *const buffer = static_cast<large_memory *>(raw);

66 WJR_MALLOC void *__small_allocate(size_t n, stack_context &context) noexcept {
67 if (WJR_UNLIKELY(static_cast<size_t>(m_cache.end - m_cache.ptr) < n)) {
68 if (auto ptr = __small_reallocate(); ptr)
69 context.m_ptr = static_cast<std::byte *>(ptr);
70 }
71
72 WJR_ASSERT_ASSUME_L2(m_cache.ptr != nullptr);
73
74 auto *const ret = assume_aligned<mem::default_new_alignment>(m_cache.ptr);
75 m_cache.ptr += align_up(n, mem::default_new_alignment);
76 return ret;
77 }

79 void __small_deallocate(const stack_context &context) noexcept {
80 const auto curr = m_idx;
81 m_idx = context.m_idx;
82 m_cache.ptr = context.m_ptr;
83 // Fast path.
84 if (WJR_LIKELY(m_idx == curr)) {
85 return;
86 }
87
88 m_cache.end = m_ptr[m_idx].end;
89 if (WJR_UNLIKELY(m_size - m_idx >= bufsize))
90 __small_redeallocate();
91 }
