59WJR_INTRINSIC_CONSTEXPR uint64_t fallback_subc_n(uint64_t *dst, const uint64_t *src0,
60 const uint64_t *src1, size_t n,
61 uint64_t c_in) noexcept {
62 size_t m = n / 4;
63
64 for (size_t i = 0; i < m; ++i) {
65 dst[0] = subc(src0[0], src1[0], c_in, c_in);
66 dst[1] = subc(src0[1], src1[1], c_in, c_in);
67 dst[2] = subc(src0[2], src1[2], c_in, c_in);
68 dst[3] = subc(src0[3], src1[3], c_in, c_in);
69
70 dst += 4;
71 src0 += 4;
72 src1 += 4;
73 }
74
75 n &= 3;
76 if (WJR_UNLIKELY(n == 0)) {
77 return c_in;
78 }
79
80 dst += n;
81 src0 += n;
82 src1 += n;
83
84 switch (n) {
85 case 3: {
86 dst[-3] = subc(src0[-3], src1[-3], c_in, c_in);
87 WJR_FALLTHROUGH;
88 }
89 case 2: {
90 dst[-2] = subc(src0[-2], src1[-2], c_in, c_in);
91 WJR_FALLTHROUGH;
92 }
93 case 1: {
94 dst[-1] = subc(src0[-1], src1[-1], c_in, c_in);
95 WJR_FALLTHROUGH;
96 }
97 default: {
98 break;
99 }
100 }
101
102 return c_in;
103}

186WJR_INTRINSIC_CONSTEXPR20 ssize_t abs_subc_n(uint64_t *dst, const uint64_t *src0,
187 const uint64_t *src1, size_t n) noexcept {
188 WJR_ASSERT_ASSUME(n >= 1);
189 WJR_ASSERT_L2(WJR_IS_SAME_OR_SEPARATE_P(dst, n, src0, n));
190 WJR_ASSERT_L2(WJR_IS_SAME_OR_SEPARATE_P(dst, n, src1, n));
191
192 size_t idx = reverse_find_not_n(src0, src1, n);
193
194 if (WJR_UNLIKELY(idx != n)) {
195 set_n(dst + idx, 0, n - idx);
196
197 if (WJR_UNLIKELY(idx == 0)) {
198 return 0;
199 }
200 }
201
202 WJR_ASSUME(idx >= 1);
203
204 uint64_t hi;
205 WJR_ASSUME(src0[idx - 1] != src1[idx - 1]);
206 const bool overflow = sub_overflow(src0[idx - 1], src1[idx - 1], hi);
207
208 if (overflow) {
209 std::swap(src0, src1);
210 hi = -hi;
211 }
212
213 do {
214 if (WJR_UNLIKELY(idx == 1)) {
215 dst[0] = hi;
216 break;
217 }
218
219 hi -= subc_n(dst, src0, src1, idx - 1);
220 dst[idx - 1] = hi;
221 } while (0);
222
223 return overflow ? -1 : 1;
224}

227 const uint64_t *src1, size_t n) noexcept {
228 WJR_ASSERT_ASSUME(n >= 1);
229 WJR_ASSERT_L2(WJR_IS_SAME_OR_SEPARATE_P(dst, n, src0, n));
230 WJR_ASSERT_L2(WJR_IS_SAME_OR_SEPARATE_P(dst, n, src1, n));
231
232 size_t idx = reverse_find_not_n(src0, src1, n);
233

235 set_n(dst + idx, 0, n - idx);
236
237 if (WJR_UNLIKELY(idx == 0)) {
238 return 0;
239 }
240 }

242 WJR_ASSUME(idx >= 1);
243

245 WJR_ASSUME(src0[idx - 1] != src1[idx - 1]);
246 const bool overflow = sub_overflow(src0[idx - 1], src1[idx - 1], hi);
247 WJR_ASSUME(hi != 0);
248
249 if (overflow) {
250 std::swap(src0, src1);
251 hi = -hi;
252 }
253
254 ssize_t ret = __fast_from_unsigned(idx);
255 WJR_ASSUME(ret >= 1);
256

263 dst[idx - 1] = hi;
264 } else {

266 }
267 } while (0);

270 return overflow ? -ret : ret;

326WJR_INTRINSIC_CONSTEXPR20 ssize_t abs_subc_s_pos(uint64_t *dst, const uint64_t *src0, size_t n,
327 const uint64_t *src1, size_t m) noexcept {
328 WJR_ASSERT_ASSUME(m >= 1);
329 WJR_ASSERT_ASSUME(n >= m);
330
331 if WJR_BUILTIN_CONSTANT_CONSTEXPR (WJR_BUILTIN_CONSTANT_P_TRUE(n == m)) {
332 return abs_subc_n_pos(dst, src0, src1, m);
333 }
334
335 if WJR_BUILTIN_CONSTANT_CONSTEXPR (WJR_BUILTIN_CONSTANT_P_TRUE(n - m <= 1)) {
336 do {
337 if (n == m) {
338 break;
339 }
340
341 if (WJR_UNLIKELY(src0[m] == 0)) {
342 dst[m] = 0;
343 break;
344 }
345
346 uint64_t hi = src0[m];
347 hi -= subc_n(dst, src0, src1, m);
348 ssize_t ret = __fast_from_unsigned(m + 1);
349
350 if (WJR_LIKELY(hi != 0)) {
351 dst[m] = hi;
352 } else {
353 --ret;
354 }
355
356 WJR_ASSUME(ret > 0);
357 return ret;
358 } while (0);
359
360 return abs_subc_n_pos(dst, src0, src1, m);
361 }
362
363 size_t idx = reverse_replace_find_not(dst + m, src0 + m, n - m, 0, 0);
364
365 if (WJR_UNLIKELY(idx == 0)) {
366 return abs_subc_n_pos(dst, src0, src1, m);
367 }
368
369 (void)subc_s(dst, src0, m + idx, src1, m);
370 ssize_t ret = __fast_from_unsigned(m + idx);
371 WJR_ASSUME(ret >= 2);
372 ret -= dst[m + idx - 1] == 0;
373 WJR_ASSUME(ret >= 1);
374 return ret;
375}

378WJR_INTRINSIC_CONSTEXPR20 ssize_t abs_subc_n(uint64_t *dst, const uint64_t *src0,
379 const uint64_t *src1, size_t n, uint64_t &c_out,
380 uint64_t cf0, uint64_t cf1) noexcept {

390 cf = cf0 - cf1;

395 }
