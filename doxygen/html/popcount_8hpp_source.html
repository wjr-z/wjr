26template <typename T, T seed>
27class de_bruijn {
28public:
29 constexpr static uint8_t digits = std::numeric_limits<T>::digits;
30 constexpr static uint8_t mv = digits == 32 ? 27 : 58;
31 constexpr de_bruijn() noexcept : lookup(), lookupr() { initialize(); }
32
33 constexpr int get(T idx) const noexcept { return lookup[(idx * seed) >> mv]; }
34 constexpr int getr(T idx) const noexcept { return lookupr[(idx * seed) >> mv]; }
35
36private:
37 constexpr void initialize() noexcept {
38 for (uint8_t i = 0; i < digits; ++i) {
39 const auto idx = (seed << i) >> mv;
40 lookup[idx] = i;
41 lookupr[idx] = i == 0 ? 0u : digits - i;
42 }
43 }
44
45 uint8_t lookup[digits];
46 uint8_t lookupr[digits];
47};

57WJR_CONST WJR_INTRINSIC_CONSTEXPR int fallback_popcount(T x) noexcept {
58 constexpr auto nd = std::numeric_limits<T>::digits;
59 if constexpr (nd < 32) {
60 return fallback_popcount(static_cast<uint32_t>(x));
61 } else {
62 if constexpr (nd == 32) {
63 x -= (x >> 1) & 0x5555'5555;
64 x = (x & 0x3333'3333) + ((x >> 2) & 0x3333'3333);
65 x = (x + (x >> 4)) & 0x0f0f'0f0f;
66 return (x * 0x0101'0101) >> 24;
67 } else {
68 x -= (x >> 1) & 0x5555'5555'5555'5555;
69 x = (x & 0x3333'3333'3333'3333) + ((x >> 2) & 0x3333'3333'3333'3333);
70 x = (x + (x >> 4)) & 0x0f0f'0f0f'0f0f'0f0f;
