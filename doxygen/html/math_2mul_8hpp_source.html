36WJR_INTRINSIC_CONSTEXPR20 uint64_t fallback_mul64(uint64_t a, uint64_t b, uint64_t &hi) noexcept {
37 uint64_t ah = a >> 32;
38 uint64_t al = a & 0xFFFFFFFF;
39 uint64_t bh = b >> 32;
40 uint64_t bl = b & 0xFFFFFFFF;
41
42 uint64_t rh = ah * bh;
43 uint64_t rm0 = ah * bl;
44 uint64_t rm1 = al * bh;
45 uint64_t rl = al * bl;

47 add_128(rl, rh, rl, rh, rm0 << 32, rm0 >> 32);
48 add_128(rl, rh, rl, rh, rm1 << 32, rm1 >> 32);

50 hi = rh;
