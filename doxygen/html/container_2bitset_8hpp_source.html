482template <size_t N>
483template <typename Char, typename Traits>
484constexpr bitset<N>::bitset(std::basic_string_view<Char, Traits> str, size_t pos, size_t n,
485 Char zero, Char one) {
486 str = str.substr(pos, n);
487 auto count = std::min<size_t>(N, str.size());
488 WJR_ASSUME(count <= N);
489
490 size_t xpos = 0;
491
492 if (count != 0) {
493 const auto *ptr = str.data();
494 ptr += count;
495
496 while (count >= bits) {
497 uint64_t val = 0;
498 for (size_t i = 0; i < bits; ++i) {
499 const auto ch = ptr[-i - 1];
500 if (Traits::eq(ch, one)) {
501 val |= static_cast<bit_type>(ch) << i;
502 } else if (!Traits::eq(ch, zero)) {
503 WJR_THROW(std::invalid_argument("invalid bitset char"));
504 }
505 }
506
507 ptr -= bits;
508 count -= bits;
509 m_data[xpos++] = val;
510 }
511
512 if (count != 0) {
513 uint64_t val = 0;
514
515 for (size_t i = 0; i < count; ++i) {
516 const auto ch = ptr[-i - 1];
517 if (Traits::eq(ch, one)) {
518 val |= static_cast<bit_type>(ch) << i;
519 } else if (!Traits::eq(ch, zero)) {
520 WJR_THROW(std::invalid_argument("invalid bitset char"));
521 }
522 }
523
524 m_data[xpos++] = val;
525 }
526 }
527
528 for (; xpos < bytes_size; ++xpos) {
529 m_data[xpos] = 0;
530 }

534constexpr bitset<N> &bitset<N>::operator<<=(size_t offset) noexcept {
535 if (WJR_UNLIKELY(offset == 0)) {
536 return *this;
537 }
538
539 bit_type low = m_data[0];
540 m_data[0] <<= offset;
541
542 if constexpr (bytes_size & 1) {
543 if constexpr (bytes_size == 1) {
544 if constexpr (!full) {
545 m_data[0] &= mask;
546 }
547 } else {
548 if constexpr (full) {
549 for (size_t i = 1; i != bytes_size; i += 2) {
550 bit_type now = m_data[i];
551 m_data[i] = shld(now, low, offset);
552 low = m_data[i + 1];
553 m_data[i + 1] = shld(low, now, offset);
554 }
555 } else {
556 for (size_t i = 1; i < bytes_size - 2; i += 2) {
557 bit_type now = m_data[i];
558 m_data[i] = shld(now, low, offset);
559 low = m_data[i + 1];
560 m_data[i + 1] = shld(low, now, offset);
561 }
562
563 bit_type now = m_data[bytes_size - 2];
564 m_data[bytes_size - 2] = shld(now, low, offset);
565 low = m_data[bytes_size - 1];
566 m_data[bytes_size - 1] = shld(low, now, offset) & mask;
567 }
568 }
569 } else {
570 for (size_t i = 1; i != bytes_size - 1; i += 2) {
571 bit_type now = m_data[i];
572 m_data[i] = shld(now, low, offset);
573 low = m_data[i + 1];
